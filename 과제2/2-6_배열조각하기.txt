#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> arr, vector<int> query) {
    vector<int> answer;
    for (int i = 0; i < query.size(); i++)
 {
       if (i % 2 == 0)   // 짝수 인덱스
       {
           arr = vector<int>(arr.begin(),arr.begin()+query[i]+1);
       }
       else// 홀수 인덱스
        {
           arr = vector<int>(arr.begin()+query[i],arr.end());
        }

 }

    answer=arr;

    return answer;
}
