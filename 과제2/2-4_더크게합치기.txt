#include <string>
#include <vector>

using namespace std;

int solution(int a, int b) {
    int answer = 0;
    
    string stra = to_string(a);
    string strb = to_string(b);
    
    string aplusb = stra+strb;
    string bplusa = strb+stra;
    
    int ab = stoi(aplusb);
    int ba = stoi(bplusa);
    
    
    if(ab>=ba)
    {
        
        return ab;
    }
    else
    {
        return ba;
        
    }
    
    return answer;
}
