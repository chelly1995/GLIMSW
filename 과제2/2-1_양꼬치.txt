#include <stdio.h>
#include <stdbool.h>
#include <stdlib.h>

int solution(int n, int k) {
    int answer = 0;
    
    if(n<10)
    {
        answer = n*12000+k*2000;
    }
    else
    {
        int a = n/10;
        
        answer = n*12000+(k-a)*2000;
    }
    
    return answer;
}
